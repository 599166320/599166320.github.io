    put用法

    put '表名','主键','列族:列名','值'

    create用法

    create '表名','列族名'

    scan用法

    查询表

    scan '表名'

    get用法

    查询具体列族

    get '表名','主键','列族名:列'

    删除表步骤

    禁用表

    disable '表名'

    删除表

    drop '表名'

    例子
